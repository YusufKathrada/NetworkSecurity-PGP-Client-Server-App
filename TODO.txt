SENDER: {sender}
RECIPIENT: {recipient}
TIMESTAMP: {timestamp}

{message}

EOF"""

# SEND
SEND_REQUEST = """SEND
SENDER: {sender}
RECIPIENT: {recipient}
TIMESTAMP: {timestamp}

{message}

EOF"""


send_header(clientsocket, RECEIVE_REQUEST.format(
    sender=sender,
    recipient=recipient,
    timestamp=datetime.datetime.now().isoformat(),
    message=message))
